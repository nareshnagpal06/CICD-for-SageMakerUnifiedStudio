# IMPORTANT: Pin scikit-learn to the container's pre-installed version.
# The sagemaker-xgboost:1.7-1 container ships scikit-learn==0.24.1 with
# compiled Cython extensions. Upgrading it (e.g. to 1.5.0) makes pip
# uninstall 0.24.1 and install a build compiled against a different
# numpy/Cython ABI, which fails at import with
# "KeyError: '__reduce_cython__'". Keeping this pin at the pre-installed
# version means pip leaves it untouched. The training script only uses
# basic sklearn.metrics functions (roc_auc_score, accuracy_score,
# f1_score) that are available in 0.24.1.
scikit-learn==0.24.1
